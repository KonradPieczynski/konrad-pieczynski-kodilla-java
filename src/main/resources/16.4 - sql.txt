SELECT I2.NAME,I.SUMMARY,I.DESCRIPTION From ISSUES I
JOIN ISSUESLISTS I2 on I2.ID = I.ISSUESLIST_ID;

SELECT U.FIRSTNAME,U.LASTNAME,I.SUMMARY,I.DESCRIPTION From ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME,U.LASTNAME, COUNT(*) AS TASK_COUNT From ISSUES I
JOIN USERS U on I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) > 0;
